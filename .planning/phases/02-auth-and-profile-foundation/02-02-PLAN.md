    **1. Add UserDefaults persistence for quiz state (ONBD-04):**

    Add a private enum for UserDefaults keys at the top of `PreferencesQuizView`:
    ```swift
    private enum QuizKeys {
        static let currentStep = "quiz_currentStep"
        static let firstName = "quiz_firstName"
        static let lastName = "quiz_lastName"
        static let phone = "quiz_phone"
        static let workAuth = "quiz_workAuth"
        static let requiresSponsorship = "quiz_requiresSponsorship"
        static let skills = "quiz_skills"
        static let jobTypes = "quiz_jobTypes"
        static let remoteOnly = "quiz_remoteOnly"
        static let resumeFileName = "quiz_resumeFileName"
    }
    ```

    Add a static helper method to clear quiz state:
    ```swift
    static func clearSavedQuizState() {
        let defaults = UserDefaults.standard
        [QuizKeys.currentStep, QuizKeys.firstName, QuizKeys.lastName,
         QuizKeys.phone, QuizKeys.workAuth, QuizKeys.requiresSponsorship,
         QuizKeys.skills, QuizKeys.jobTypes, QuizKeys.remoteOnly,
         QuizKeys.resumeFileName].forEach { defaults.removeObject(forKey: $0) }
    }
    ```

    **Restore on appear:** Add an `.onAppear` modifier to the outermost `NavigationStack` in `body`:
    ```swift
    .onAppear {
        let defaults = UserDefaults.standard
        currentStep = defaults.integer(forKey: QuizKeys.currentStep)
        firstName = defaults.string(forKey: QuizKeys.firstName) ?? ""
        lastName = defaults.string(forKey: QuizKeys.lastName) ?? ""
        phone = defaults.string(forKey: QuizKeys.phone) ?? ""
        workAuth = defaults.string(forKey: QuizKeys.workAuth) ?? "US Citizen"
        requiresSponsorship = defaults.bool(forKey: QuizKeys.requiresSponsorship)
        skills = defaults.stringArray(forKey: QuizKeys.skills) ?? []
        jobTypes = defaults.stringArray(forKey: QuizKeys.jobTypes) ?? []
        remoteOnly = defaults.bool(forKey: QuizKeys.remoteOnly)
        resumeFileName = defaults.string(forKey: QuizKeys.resumeFileName) ?? ""
        // Note: resumeData (binary) is NOT persisted — user must re-select PDF after force-quit
        // But resumeFileName is preserved so the UI shows the previous selection name
    }
    ```

    **Save on change:** Add `.onChange` modifiers after the `.onAppear`:
    ```swift
    .onChange(of: currentStep) { UserDefaults.standard.set($0, forKey: QuizKeys.currentStep) }
    .onChange(of: firstName) { UserDefaults.standard.set($0, forKey: QuizKeys.firstName) }
    .onChange(of: lastName) { UserDefaults.standard.set($0, forKey: QuizKeys.lastName) }
    .onChange(of: phone) { UserDefaults.standard.set($0, forKey: QuizKeys.phone) }
    .onChange(of: workAuth) { UserDefaults.standard.set($0, forKey: QuizKeys.workAuth) }
    .onChange(of: requiresSponsorship) { UserDefaults.standard.set($0, forKey: QuizKeys.requiresSponsorship) }
    .onChange(of: skills) { UserDefaults.standard.set($0, forKey: QuizKeys.skills) }
    .onChange(of: jobTypes) { UserDefaults.standard.set($0, forKey: QuizKeys.jobTypes) }
    .onChange(of: remoteOnly) { UserDefaults.standard.set($0, forKey: QuizKeys.remoteOnly) }
    .onChange(of: resumeFileName) { UserDefaults.standard.set($0, forKey: QuizKeys.resumeFileName) }
    ```

    Note on iOS 16 `.onChange` syntax: iOS 16 uses the old-style `.onChange(of: value) { newValue in ... }`. iOS 17+ uses the no-parameter closure. Since this project targets iOS 16+, use the old style:
    ```swift
    .onChange(of: currentStep) { newValue in UserDefaults.standard.set(newValue, forKey: QuizKeys.currentStep) }
    ```
